<project_specification>
  <project_name>Real Team Translation</project_name>

  <overview>
    A real-time transcription and translation tool for journalists. Users capture audio from meetings or calls via their browser microphone, transcribe speech in real-time using Deepgram with speaker diarization, and translate the transcript on-the-fly to a target language via OpenRouter. Transcripts are saved per session and can be exported as Markdown files.
  </overview>

  <technology_stack>
    <frontend>
      <framework>Next.js 16 (App Router) with React 19 and TypeScript</framework>
      <styling>Tailwind CSS v4 + shadcn/ui (new-york variant, neutral base)</styling>
      <components>Lucide React icons, Sonner toasts, next-themes for dark/light mode</components>
      <fonts>Geist (sans) and Geist Mono (monospace)</fonts>
    </frontend>
    <backend>
      <runtime>Next.js 16 API Routes (App Router)</runtime>
      <database>PostgreSQL 18 with pgvector extension (Docker)</database>
      <orm>Drizzle ORM</orm>
      <auth>Better Auth (email/password, sessions)</auth>
      <realtime>Socket.io for WebSocket communication (audio streaming, transcript delivery)</realtime>
    </backend>
    <communication>
      <api>REST API for session CRUD, WebSocket (Socket.io) for real-time audio/transcript streaming</api>
      <speech_to_text>Deepgram API via WebSocket (streaming transcription with speaker diarization)</speech_to_text>
      <translation>OpenRouter via Vercel AI SDK (streamText) for real-time translation of transcript segments</translation>
    </communication>

  </technology_stack>

  <prerequisites>
    <environment_setup>
      1. PostgreSQL 18 running via Docker Compose (docker-compose.yml provided)
      2. Node.js 20+ and pnpm installed
      3. Environment variables configured in .env:
         - DEEPGRAM_API_KEY (for speech-to-text)
         - OPENROUTER_API_KEY (for translation LLM)
         - OPENROUTER_MODEL (e.g., "openai/gpt-4o-mini" or "anthropic/claude-sonnet-4-20250514")
         - POSTGRES_URL (PostgreSQL connection string)
         - BETTER_AUTH_SECRET (session encryption)
         - PORT (default 3000)
      4. Run: pnpm install
      5. Run: pnpm run db:push (apply schema to database)
      6. Run: pnpm run dev (start development server)
    </environment_setup>
  </prerequisites>

  <database_schema>
    <tables>
      <session>
        - id (text, primary key, cuid)
        - user_id (text, foreign key -> user.id, not null, cascade delete)
        - title (text, not null, default "Untitled Session")
        - status (text, not null, default "completed" — values: "recording", "paused", "completed")
        - source_language (text, default "auto-detected" — the detected language of the audio)
        - target_language (text, not null — the translation target language code, e.g., "es", "fr", "en")
        - duration_seconds (integer, default 0 — total recording duration)
        - speaker_count (integer, default 0 — number of detected speakers)
        - created_at (timestamp, not null, default now)
        - updated_at (timestamp, not null, default now)
      </session>

      <transcript_segment>
        - id (text, primary key, cuid)
        - session_id (text, foreign key -> session.id, not null, cascade delete)
        - speaker_label (text, not null — e.g., "Speaker 1", "Speaker 2")
        - original_text (text, not null — the transcribed text in source language)
        - translated_text (text — the translated text in target language, null until translated)
        - start_time (float — offset in seconds from session start)
        - end_time (float — offset in seconds from session start)
        - is_final (boolean, not null, default false — whether this is a final or interim segment)
        - created_at (timestamp, not null, default now)
        - updated_at (timestamp, not null, default now)
      </transcript_segment>

      <user_settings>
        - id (text, primary key, cuid)
        - user_id (text, foreign key -> user.id, not null, unique, cascade delete)
        - default_target_language (text, default "en" — user's preferred translation language)
        - selected_microphone_id (text — browser microphone device ID preference)
        - updated_at (timestamp, not null, default now)
      </user_settings>
    </tables>

    <indexes>
      - session: index on user_id, index on created_at (descending)
      - transcript_segment: index on session_id, index on (session_id, created_at)
      - user_settings: unique index on user_id
    </indexes>

    <relationships>
      - user 1:N session (a user has many sessions)
      - session 1:N transcript_segment (a session has many transcript segments)
      - user 1:1 user_settings (a user has one settings record)
    </relationships>
  </database_schema>

  <api_endpoints_summary>
    <sessions>
      - POST /api/sessions — Create a new transcription session
      - GET /api/sessions — List user's sessions (sorted by date desc)
      - GET /api/sessions/[id] — Get session with transcript segments
      - DELETE /api/sessions/[id] — Delete a session and its segments
    </sessions>

    <export>
      - GET /api/sessions/[id]/export — Export session transcript as Markdown
    </export>

    <settings>
      - GET /api/settings — Get user settings
      - PATCH /api/settings — Update user settings (default language, mic preference)
    </settings>

    <websocket_events>
      - Client -> Server: "audio-chunk" (binary audio data from microphone)
      - Client -> Server: "start-recording" (session_id, target_language)
      - Client -> Server: "stop-recording" (session_id)
      - Client -> Server: "pause-recording" (session_id)
      - Client -> Server: "resume-recording" (session_id)
      - Server -> Client: "transcript-interim" (interim transcription result with speaker label)
      - Server -> Client: "transcript-final" (final transcription result with speaker label and timestamps)
      - Server -> Client: "translation" (translated text for a final segment)
      - Server -> Client: "recording-status" (current recording state: active, paused, stopped)
      - Server -> Client: "error" (error message for Deepgram/translation failures)
    </websocket_events>

    <existing_endpoints_to_keep>

      - GET /api/diagnostics — Health check (keep as-is)
      - POST /api/chat — OpenRouter chat (keep as-is, used by existing chat page)
    </existing_endpoints_to_keep>
  </api_endpoints_summary>
